# 1. Repository auf GitHub erstellen (ohne .git initialisieren!)
# Gehe zu https://github.com/new
# Repository Name: timetracker-website
# Public aktivieren

# 2. Im Verzeichnis initialisieren
cd c:\Users\simsi\Desktop\TimeTracker\website
git init
git add .
git commit -m "Initial commit: TimeTracker landing page"

# 3. Remote hinzufügen und pushen
git remote add origin https://github.com/SimonKroisi/timetracker-website.git
git branch -M main
git push -u origin main

# 4. GitHub Pages aktivieren:
# Settings → Pages → Source: Deploy from branch → main → /root
# Fertig! Deine Website ist unter https://SimonKroisi.github.io/timetracker-website erreichbar
